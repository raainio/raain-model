ğŸ“– Quick Start

import {RainComputationMap, RainNode, RadarNode} from 'raain-model';

// Create a radar node
const radarNode = new RadarNode({
    id: 'radar1',
    latitude: 48.8566,
    longitude: 2.3522,
    name: 'Paris Radar',
    team: null
});

// Create a rain node
const rainNode = new RainNode({
    id: 'rain1',
    name: 'Paris Rain',
    team: null,
    radars: [radarNode]
});

// Process rain data
const computation = new RainComputationMap({
    id: 'comp1',
    date: new Date(),
    isReady: true,
    map: [] // Your rain measurement data here
});
